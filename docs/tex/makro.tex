\subsection{Model IS}

\begin{equation}
    \mathrm{AE} = C + I + G = a + c(1 - t)Y - c T + c B + \bar{I} - b i + G,
    \end{equation}
gdzie:
\begin{itemize}
    \item \(AE\) - Popyt globalny (agregatowy popyt)
    \item \(C\) - Konsumpcja
    \item \(I\) - Inwestycje
    \item \(G\) - Wydatki rządowe
\end{itemize}


\begin{equation}
    I =  I_0 - b \cdot i
\end{equation}
gdzie:
\begin{itemize}
    \item \(I_0\) - Inwestycje niezależne od stopy procentowej
    \item \(b\) - Wrażliwość inwestycji na stopę procentową
    \item \(i\) - Stopa procentowa
\end{itemize}


\begin{equation}
    Y = \bar{A} + c(1 - t)Y - b i,
    \end{equation}

    \begin{equation}
        Y = \alpha(\bar{A} - b i),
        \end{equation}
        
        \begin{equation}
        \alpha = \frac{1}{1 - c(1 - t)},
        \end{equation}

\begin{equation}
    \bar{A} = a - c T + c B + \bar{I} + G
    \end{equation}

    gdzie:
    \begin{itemize}
        \item \(\bar{A}\) - wydatki autonomiczne
        \item \(a\) - Autonomiczna konsumpcja
        \item \(c\) - Krańcowa skłonność do konsumpcji
        \item \(T\) - Podatki
        \item \(B\) - Transfery od rządu
        \item \(\bar{I}\) - Inwestycje autonomiczne
        \item \(G\) - Wydatki rządowe
    \end{itemize}
